19  /**
20   * Class represents a component that draws {@link BarChart}.
21   * @author Vilim StaroveÅ¡ki
22   *
23   */
24  public class BarChartComponent extends JComponent {
25  
26  	/**
27  	 * Generated serial version UID.
28  	 */
29  	private static final long serialVersionUID = -2290688286165650352L;
30  
31  	/**
32  	 * Chart this component draws.
33  	 */
34  	private BarChart chart;
35  	
36  	/**
37  	 * Creates new {@link BarChartComponent}
38  	 * @param chart Chart this component draws.
39  	 */
40  	public BarChartComponent(BarChart chart) {
41  
42  		this.chart = chart;
43  		Border current = this.getBorder();
44  		Border empty = new EmptyBorder(5, 5, 5, 5);
45  		if (current == null) {
46  		    this.setBorder(empty);
47  		}
48  		else {
49  		    this.setBorder(new CompoundBorder(empty, current));
50  		}
51  	}
52  	
53  	@Override
54  	protected void paintComponent(Graphics g) {
55  		
56  		Graphics2D g2d = (Graphics2D) g;
57  		FontMetrics fm = g.getFontMetrics();
58  		Insets insets = getInsets();
59  		Dimension size = getSize();
60  		g2d.setColor(Color.BLACK);
61  
62  		//drawing first frame; contains x and y axis descriptions
63  		Rectangle frameForDrawing = new Rectangle(insets.left, insets.top, 
64  													size.width-insets.right-insets.left, 
65  													size.height-insets.top-insets.bottom);
66  		
67  //		g2d.drawLine(frameForDrawing.x, frameForDrawing.y, frameForDrawing.x + frameForDrawing.width, frameForDrawing.y);
68  		
69  		AffineTransform defaultAt = g2d.getTransform();
70  		g2d.setTransform(defaultAt);
71  		
72  		String xDescription = chart.getxDes();
73  		int lengthOfX = fm.stringWidth(xDescription);
74  		g2d.drawString(xDescription, frameForDrawing.x + frameForDrawing.width/2 - lengthOfX/2, frameForDrawing.y + frameForDrawing.height);
75  		
76  		AffineTransform at = AffineTransform.getQuadrantRotateInstance(3);
77  		g2d.setTransform(at);
78  		
79  		String yDescription = chart.getyDes();
80  		int lengthOfY = fm.stringWidth(yDescription);
81  		g2d.drawString(yDescription, frameForDrawing.y - frameForDrawing.height/2 - lengthOfY/2, fm.getHeight());
82  		g2d.setTransform(defaultAt);
83  		
84  		//drawing second frame, that is inside first frame; contains numbers for axis
85  		int insetsForSecondFrame = fm.getHeight();
86  		Rectangle secondFrame = new Rectangle(frameForDrawing.x + insetsForSecondFrame, frameForDrawing.y + insetsForSecondFrame, 
87  												frameForDrawing.width - 2*insetsForSecondFrame, 
88  												frameForDrawing.height - 2*insetsForSecondFrame);
89  		
90  //		g2d.drawLine(secondFrame.x, secondFrame.y, secondFrame.width + secondFrame.x, secondFrame.y);
91  //		g2d.drawLine(secondFrame.x, secondFrame.y, secondFrame.x, secondFrame.height + secondFrame.y);
92  //		g2d.drawLine(secondFrame.x, secondFrame.height + secondFrame.y, secondFrame.x + secondFrame.width, secondFrame.height + secondFrame.y);
93  		
94  		
95  		int widthOfYMax = fm.stringWidth(String.valueOf(chart.getyMax()));
96  		int widthOfYMin = fm.stringWidth(String.valueOf(chart.getyMin()));
97  		int maxWidth = Math.max(widthOfYMax, widthOfYMin);
98  		int yValues = (chart.getyMax()-chart.getyMin()) / chart.getStep() + 1;
99  		//drawing third frame
100 		int insetForThirdFrameBottomTop = fm.getHeight();
101 		int insetsForThirdFrameLR = maxWidth + 5;
102 		Rectangle thirdFrame = new Rectangle(secondFrame.x + insetsForThirdFrameLR, 
103 												secondFrame.y + insetForThirdFrameBottomTop, 
104 												secondFrame.width - 2*insetsForThirdFrameLR, 
105 												secondFrame.height - 2*insetForThirdFrameBottomTop);
106 		
107 //		g2d.drawLine(thirdFrame.x, thirdFrame.y, thirdFrame.width + thirdFrame.x, thirdFrame.y);
108 //		g2d.drawLine(thirdFrame.x, thirdFrame.y, thirdFrame.x, thirdFrame.height + thirdFrame.y);
109 //		g2d.drawLine(thirdFrame.x, thirdFrame.height + thirdFrame.y, thirdFrame.x + thirdFrame.width, thirdFrame.height + thirdFrame.y);
110 		
111 		 g2d.setFont(g2d.getFont().deriveFont(Font.BOLD));
112 		
113 		int spaceBetweenVertical = (thirdFrame.height + fm.getHeight()) / yValues - fm.getHeight();
114 		int[] valuesOnY = new int[yValues];
115 		for(int i = 0; i < yValues; i++) {
116 			valuesOnY[i] = i*chart.getStep();
117 		}
118 		for(int i = 0; i < yValues; i++) {
119 			g2d.drawString(String.valueOf(valuesOnY[i]), 
120 					secondFrame.x + (maxWidth - fm.stringWidth(String.valueOf(valuesOnY[i]))), 
121 					thirdFrame.y + thirdFrame.height + fm.getHeight()/2 - i*(fm.getHeight() + spaceBetweenVertical));
122 		}
123 		
124 		int[] valuesOnX = new int[chart.getValues().size()];
125 		for(int i = 0; i < chart.getValues().size(); i++) {
126 			valuesOnX[i] = chart.getValues().get(i).getX();
127 		}
128 		Arrays.sort(valuesOnX);
129 		int availableSpaceForXValue = thirdFrame.width / valuesOnX.length;
130 		for(int i = 0; i < valuesOnX.length; i++) {
131 			int offset = (availableSpaceForXValue - fm.stringWidth(String.valueOf(valuesOnX[i]))) /2;
132 			g2d.drawString(String.valueOf(valuesOnX[i]), 
133 					thirdFrame.x + i*availableSpaceForXValue + offset, 
134 					thirdFrame.y+thirdFrame.height + fm.getHeight());
135 		}
136 		
137 		//setting the actual chart picture
138 		//frame is 'thirdFrame'
139 		for(int i = 0; i < yValues ; i++) {
140 			if(i == 0) {
141 				g2d.setColor(Color.GRAY);
142 			}
143 			else {
144 				g2d.setColor(Color.YELLOW);
145 			}
146 			g2d.drawLine(thirdFrame.x - 3, 
147 						 thirdFrame.y + thirdFrame.height - i*(fm.getHeight() + spaceBetweenVertical) + 3,
148 						 thirdFrame.x + thirdFrame.width, 
149 						 thirdFrame.y + thirdFrame.height - i*(fm.getHeight() + spaceBetweenVertical) + 3);
150 		}
151 		for(int i = 0; i <= valuesOnX.length; i++) {
152 			if(i == 0) {
153 				g2d.setColor(Color.GRAY);
154 			}
155 			else {
156 				g2d.setColor(Color.YELLOW);
157 			}
158 			g2d.drawLine(thirdFrame.x + i*availableSpaceForXValue, 
159 						 thirdFrame.y + thirdFrame.height + 3, 
160 						 thirdFrame.x + i*availableSpaceForXValue, 
161 						 thirdFrame.y);
162 		}
163 		g2d.setColor(Color.ORANGE);
164 		int i = 0;
165 		XYValue values[] = new XYValue[chart.getValues().size()];
166 		for(XYValue value : chart.getValues()) {
167 			int heightOfColumn = (int) Math.round((double)value.getY() / chart.getStep() * (fm.getHeight() + spaceBetweenVertical));
168 			Rectangle rect = new Rectangle(thirdFrame.x + i*availableSpaceForXValue, 
169 											thirdFrame.y + thirdFrame.height - heightOfColumn + 2, 
170 											availableSpaceForXValue - 2, 
171 											heightOfColumn);
172 			g2d.draw(rect);
173 			g2d.fill(rect);
174 			values[i] = value;
175 			i++;
176 		}
177 		
178 		g2d.setColor(Color.LIGHT_GRAY);
179 		for(i = 0; i < values.length; i++) {
180 			boolean hasShadow = false;
181 			int diference = 0;
182 			if(i == values.length - 1) {
183 				hasShadow = true;
184 				diference = values[i].getY();
185 			}
186 			else if(values[i].getY() > values[i+1].getY()) {
187 				hasShadow = true;
188 				diference = values[i].getY() - values[i+1].getY();
189 			}
190 			
191 			if(hasShadow) {
192 				
193 				int heightOfColumn = (int) Math.round((double)values[i].getY() / chart.getStep() * (fm.getHeight() + spaceBetweenVertical));
194 				int diferenceHeight = (int) Math.round((double)diference / chart.getStep() * (fm.getHeight() + spaceBetweenVertical));
195 				Rectangle rect = new Rectangle(thirdFrame.x + (i+1)*availableSpaceForXValue, 
196 												thirdFrame.y + thirdFrame.height - heightOfColumn + 5, 
197 												2, 
198 												diferenceHeight - 5);
199 				g2d.draw(rect);
200 				g2d.fill(rect);
201 			}
202 		}
203 	}
204 }
